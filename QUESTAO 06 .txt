notaUm : real
notaDois : real
media : real

Inicio
escreva("Digite sua primeira nota: ")
leia(notaUm)

escreva("Digite sua segunda nota: ")

media <- (notaUm + notaDois)/2

escreval("Sua média é:", media)

se(Media >= 6) então
escreval(" Parabens, você foi aprovado!")
fimse

se (Media < 4) entao
escreval("O aluno está reprovado.")
fimse

se(media = 4) entao
escreva("O aluno esta em recuperaçao")
